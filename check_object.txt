 funksion checkon nje list filesh/direktorish dhe afishon :

 X eshte file

Y eshte direktori

Z eshte file

#!/bin/bash


function afisho(){


a=0

b=0


for i in $@

do

if [ -d $i ]

then

echo "$i eshte directory"

let a=$a+1

else

echo "$i eshte file"

let b=$b+1

fi

done


echo "Direktori : $a"

echo "File: $b"

}


afisho $@
